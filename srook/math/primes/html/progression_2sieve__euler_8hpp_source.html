44 template <template <class...> class Container = std::vector>
45 class sieve_euler {
46  template <class... Ts>
47  struct container_type : type_constant<Container<Ts...>> {};
48 public:
58  template <class T>
59  SROOK_FORCE_INLINE SROOK_DEDUCED_TYPENAME enable_if<is_integral<T>::value, Container<T>>::type
60  operator()(T n)
61  {
62  if (n < 2) return {};
63  return sieve(n);
64  }
65 private:
66  template <class T>
67  SROOK_FORCE_INLINE Container<T> sieve(T n)
68  {
69  Container<T> rs;
70  rs.reserve(static_cast<std::size_t>(srook::math::primes::counting::legendre_compute(n)));
71  std::vector<bool> is_prime(n, true);
72  is_prime[0].flip(); is_prime[1].flip();
73  std::vector<T> smallest_fp(n); // store smallest prime factor of number
74 
75  for (std::size_t i = 2; i < static_cast<std::size_t>(n); ++i) {
76  if (is_prime[i]) {
77  rs.push_back(static_cast<T>(i));
78  smallest_fp[i] = static_cast<T>(i);
79  }
80  for (std::size_t j = 0; j < rs.size() && static_cast<std::size_t>(i * rs[j]) < static_cast<std::size_t>(n) && rs[j] <= smallest_fp[i]; ++j) {
81  is_prime[i * rs[j]] = false;
82  smallest_fp[i * rs[j]] = rs[j];
83  }
84  }
85  return rs;
86  }
87 };
