44 template <template <class...> class Container = std::vector>
45 class sieve_eratosthenese {
46  template <class... Ts>
47  struct container_type : type_constant<Container<Ts...>> {};
48 public:
59  template <class T>
60  SROOK_FORCE_INLINE SROOK_DEDUCED_TYPENAME enable_if<is_integral<T>::value, Container<T>>::type
61  operator()(T n)
62  {
63  if (n < 2) return {};
64 
65  typedef T value_type;
66  Container<value_type> rs;
67  // By using the prime number theorem, the memory area is reserved in advance.
68  rs.reserve(static_cast<std::size_t>(srook::math::primes::counting::legendre_compute(n)));
69 
70  // Find all primes upto square root of `n` and store these primes in vector `rs`.
71  const auto limit = static_cast<value_type>(std::floor(std::sqrt(n)) + 1);
72  auto oiter = sieve(limit, std::back_inserter(rs));
73 
74  // Divide the range [0..n-1] in different segments.
75  for (std::size_t low = limit, high = limit << 1; low < static_cast<std::size_t>(n); low += limit) {
76  std::vector<bool> is_prime(limit + 1, true);
77  for (const value_type& prime : rs) {
78  // Find the minimum number in [low..high] that is a multiple of prime (divisible by prime).
79  auto lolim = static_cast<std::size_t>(std::floor(low / prime)) * prime;
80  if (lolim < low) lolim += prime;
81  for (std::size_t i = lolim; i < high; i += prime) {
82  is_prime[i - low] = false;
83  }
84  }
85  for (std::size_t i = low; i < high; ++i) {
86  if (is_prime[i - low]) *oiter++ = static_cast<value_type>(i);
87  }
88  high += limit;
89  if (high >= static_cast<std::size_t>(n)) high = n;
90  }
91  return rs;
92  }
93 private:
94  template <class T, class OutputIterator>
95  SROOK_FORCE_INLINE OutputIterator sieve(T n, OutputIterator oiter)
96  {
97  std::vector<bool> prime(n, true);
98  prime[0].flip();
99  for (std::size_t i = 0, end = static_cast<std::size_t>(srook::math::sqrt(n)); i < end; ++i) {
100  if (prime[i]) {
101  for (std::size_t j = ((i + 1) << 1) - 1; j < static_cast<std::size_t>(n); j += i + 1) prime[j] = false;
102  }
103  }
104  for (std::size_t i = 1; i < static_cast<std::size_t>(n); ++i) {
105  if (prime[i]) *oiter++ = static_cast<T>(i + 1);
106  }
107  return oiter;
108  }
109 };
