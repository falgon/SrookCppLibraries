69  if (n < 3) return rs;

75  auto inserter = std::back_inserter(rs);
76  sieve(n, [&inserter](value_type p){ *inserter++ = p; });

80  template <class T, class F>
81  SROOK_FORCE_INLINE SROOK_DEDUCED_TYPENAME enable_if<is_invocable<SROOK_DEDUCED_TYPENAME decay<F>::type, T&>::value>::type
82  sieve(const T n, F&& f)
83  {
84  std::vector<bool> is_prime(n, false);
85  const std::size_t sqrtn = srook::math::sqrt(n);
86  std::size_t in = 0;
87 
88  for (std::size_t i = 1; i <= 5; i += 4) {
89  for (std::size_t j = i; j <= sqrtn; j += 6) {
90  for (std::size_t k = 1; k <= sqrtn && (in = (k * k << 2) + j * j) <= n; ++k) {
91  is_prime[in].flip();
92  }

96  }

100  for (std::size_t k = 1; k <= sqrtn && (in = 3 * k * k + j * j) <= n; k += 2) {
101  is_prime[in].flip();
102  }
103  for (std::size_t k = j + 1; k <= sqrtn && (in = 3 * k * k - j * j) <= n; k += 2) {
104  is_prime[in].flip();
105  }
106  }
107  }
108  for (std::size_t i = 3; i <= sqrtn; i += 6) {
109  for (std::size_t j = 1; j <= 2; ++j) {
110  for (std::size_t k = j; k <= sqrtn && (in = (k * k << 2) + i * i) <= n; k += 3) {
111  is_prime[in].flip();
112  }
113  }
114  }

116  if (is_prime[i]) {

118  is_prime[j] = false;
119  }
120  }
121  }

123  if (is_prime[i]) f(i);
124  }
125  }
