45 public:
56  template <class T>
57  SROOK_FORCE_INLINE std::size_t operator()(T n)
58  {
59  if (n < 2) return {};
60  return sieve(n);
61  }
62 private:
63  SROOK_FORCE_INLINE std::size_t sieve(std::size_t n)
64  {
65  std::vector<bool> prime(n, true);
66  prime[0].flip();
67  for (std::size_t i = 0, end = static_cast<std::size_t>(srook::math::sqrt(n)); i < end; ++i) {
68  if (prime[i]) {
69  for (std::size_t j = ((i + 1) << 1) - 1; j < n; j += i + 1) prime[j] = false;
70  }
71  }
72  std::size_t x = 0;
73  for (std::size_t i = 1; i < n; ++i) {
74  if (prime[i]) ++x;
75  }
76  return x;
77  }
78 };
